**Priority**: 🟡 Medium

**Effort**: Medium (3-4 hours)
**Implementation**:
```typescript
// Add validation after parsing
const warnings: string[] = [];
const errors: string[] = [];

grid.forEach((rowCells, rowOffset) => {
  const summary = rowCells[1]; // Assuming column 1 is summary
  if (!summary || summary.trim() === '') {
    errors.push(`Row ${rowOffset + 1}: 제목 필수`);
  }

  const type = rowCells[0];
  if (type && !['Epic', 'Task', 'e', 't'].includes(type.toLowerCase())) {
    warnings.push(`Row ${rowOffset + 1}: 유효하지 않은 유형 "${type}"`);
  }
});

if (errors.length > 0) {
  toast.error(`❌ 붙여넣기 실패\n${errors.join('\n')}`);
  return; // Block paste
}
```
